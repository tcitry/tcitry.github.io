                                        kubectl 常用命令大全

                                        This article is extracted from the chat log with AI. Please identify it with caution.

                                        1. 基础与配置相关命令#

                                        1.1 上下文与命名空间管理#

                                        # 查看当前上下文
kubectl config current-context

# 查看所有上下文
kubectl config get-contexts

# 切换上下文
kubectl config use-context my-context

# 查看集群信息
kubectl config view --minify

# 设置默认命名空间（为当前上下文）
kubectl config set-context --current --namespace=my-namespace

# 查看当前命名空间
kubectl config view --minify --output 'jsonpath={..namespace}'

                                        要查看 Kubernetes (k8s) 中的所有命名空间，最核心的命令是 kubectl get namespaces 或它的简写 kubectl get ns。

                                        操作命令说明与示例
                                        查看所有命名空间kubectl get namespaces查看列表
                                        kubectl get nsnamespaces 的简写形式
                                        查看更多详情kubectl get ns -o wide显示更详细的列信息
                                        kubectl get ns --show-labels显示命名空间的标签
                                        查看特定命名空间kubectl describe namespace <名称>获取某个命名空间的详细配置和资源配额等信息

                                        了解以下信息能帮你更好地使用命令：

                                        • 默认命名空间：Kubernetes 初始化后通常包含 default、kube-system、kube-public 和 kube-node-lease 等系统命名空间。
                                        • 跨命名空间操作：大多数 kubectl 命令默认针对 default 命名空间。要操作其他命名空间的资源，可以使用 -n 或 --namespace 标志，例如 kubectl get pods -n kube-system。
                                        • 查看所有命名空间的资源：如果想一次性查看所有命名空间下的某种资源（如所有 Pod），可以使用 --all-namespaces 或 -A 标志，例如 kubectl get pods -A。

kubectl version

kubectl cluster-info dump

# 查看资源 API 版本
kubectl api-versions

# 查看 API 资源

                                        1.3 自动补全配置#

                                        # 启用 bash 自动补全
source <(kubectl completion bash)

# 永久启用 bash 自动补全
echo "source <(kubectl completion bash)" >> ~/.bashrc

# 启用 zsh 自动补全
source <(kubectl completion zsh)

# 启用 fish 自动补全
kubectl completion fish | source

                                        2. 资源查看命令#

                                        2.1 通用查看命令#

                                        # 查看所有命名空间的 Pod
kubectl get pods -A

# 查看指定命名空间的 Pod
kubectl get pods -n my-namespace

# 查看所有资源（当前命名空间）
kubectl get all

kubectl get pod my-pod -o json

# 自定义列输出
kubectl get pods -o custom-columns="NAME:.metadata.name,STATUS:.status.phase,NODE:.spec.nodeName"

# 按名称排序
kubectl get pods --sort-by=.metadata.name

# 监视资源变化
kubectl get pods -w

                                        3. Pod 相关常用命令#

                                        3.1 Pod 状态查看#

                                        # 查看 Pod 状态
kubectl get pods -o wide

# 查看 Pod 详细信息
kubectl describe pod my-pod

# 查看 Pod 运行在哪个节点
kubectl get pod my-pod -o wide

# 查看 Pod 的 YAML 定义
kubectl get pod my-pod -o yaml

# 查看 Pod 标签
kubectl get pods --show-labels

                                        3.2 Pod 日志查看#

                                        # 查看 Pod 日志
kubectl logs my-pod

# 实时查看日志（类似 tail -f）
kubectl logs -f my-pod

# 查看之前容器的日志（容器崩溃时）
kubectl logs --previous my-pod

# 查看多容器 Pod 的特定容器日志
kubectl logs -f my-pod -c my-container

# 查看最近一段时间内的日志
kubectl logs --since=1h my-pod

# 查看指定时间范围内的日志
kubectl logs --since-time="2023-01-01T00:00:00Z" my-pod

                                        3.3 容器执行与调试#

                                        # 进入容器执行命令
kubectl exec -it my-pod -- /bin/bash

# 进入指定容器（多容器 Pod）
kubectl exec -it my-pod -c my-container -- /bin/sh

# 在容器中执行单条命令
kubectl exec my-pod -- ls -la

# 调试容器（使用 busybox 等工具）
kubectl debug -it my-pod --image=busybox --target=my-pod

                                        3.4 端口转发#

                                        # 端口转发到 Pod
kubectl port-forward my-pod 8080:80

# 端口转发到 Service
kubectl port-forward svc/my-service 8080:80

# 端口转发到 Deployment
kubectl port-forward deployment/my-deployment 8080:80

# 在后台运行端口转发
kubectl port-forward my-pod 8080:80 &

                                        3.5 Pod 生命周期管理#

                                        # 删除 Pod
kubectl delete pod my-pod

# 强制删除 Pod
kubectl delete pod my-pod --force --grace-period=0

# 删除所有失败的 Pod
kubectl delete pods --field-selector=status.phase=Failed

# 重启 Pod（通过删除重建）
kubectl delete pod my-pod && kubectl apply -f pod.yaml

                                        4. 工作负载管理#

                                        4.1 Deployment 管理#

                                        # 创建 Deployment
kubectl create deployment my-deploy --image=nginx:1.20

# 应用 Deployment YAML
kubectl apply -f deployment.yaml

# 查看 Deployment 状态
kubectl get deployments

# 扩容 Deployment
kubectl scale deployment my-deploy --replicas=5

# 更新 Deployment 镜像
kubectl set image deployment/my-deploy nginx=nginx:1.21

                                        4.2 滚动更新与回滚#

                                        # 查看滚动更新状态
kubectl rollout status deployment/my-deploy

# 查看更新历史
kubectl rollout history deployment/my-deploy

# 回滚到上一个版本
kubectl rollout undo deployment/my-deploy

# 回滚到指定版本
kubectl rollout undo deployment/my-deploy --to-revision=2

# 暂停滚动更新
kubectl rollout pause deployment/my-deploy

# 恢复滚动更新
kubectl rollout resume deployment/my-deploy

kubectl get jobs
kubectl describe job my-job

# CronJob 管理
kubectl get cronjobs
kubectl describe cronjob my-cronjob

                                        4.4 ReplicaSet 管理#

                                        # 查看 ReplicaSet
kubectl get replicasets

# 查看 ReplicaSet 详细信息
kubectl describe replicaset my-replicaset

# 查看 ReplicaSet 关联的 Pod
kubectl get pods --selector=app=my-app

                                        5. 配置与资源管理#

                                        5.1 资源操作命令#

                                        # 创建资源
kubectl create -f resource.yaml

# 应用资源（创建或更新）
kubectl apply -f resource.yaml

# 删除资源
kubectl delete -f resource.yaml

# 编辑资源
kubectl edit deployment/my-deploy

# 替换资源
kubectl replace -f resource.yaml

# 部分更新资源
kubectl patch deployment my-deploy -p '{"spec":{"replicas":3}}'

                                        5.2 ConfigMap 管理#

                                        # 从字面值创建 ConfigMap
kubectl create configmap my-config --from-literal=key=value

# 从文件创建 ConfigMap
kubectl create configmap my-config --from-file=config.properties

# 从目录创建 ConfigMap
kubectl create configmap my-config --from-file=config-dir/

# 查看 ConfigMap
kubectl get configmaps
kubectl describe configmap my-config

# 导出 ConfigMap 为 YAML
kubectl get configmap my-config -o yaml

                                        5.3 Secret 管理#

                                        # 创建通用 Secret
kubectl create secret generic my-secret --from-literal=password=secret

# 从文件创建 Secret
kubectl create secret generic my-secret --from-file=ssh-privatekey=~/.ssh/id_rsa

# 创建 TLS Secret
kubectl create secret tls my-tls --cert=path/to/cert --key=path/to/key

# 查看 Secret
kubectl get secrets
kubectl describe secret my-secret

                                        5.4 批量操作#

                                        # 应用目录下所有 YAML 文件
kubectl apply -f k8s-manifests/

# 递归应用目录及子目录
kubectl apply -R -f k8s-manifests/

# 删除目录下所有资源
kubectl delete -f k8s-manifests/

# 根据标签删除资源
kubectl delete all -l app=my-app

                                        6.1 Service 管理#

kubectl get services
kubectl get svc -A

# 查看 Service 详细信息
kubectl describe service my-service

# 查看 Endpoints
kubectl get endpoints
kubectl describe endpoints my-service

# 创建 Service
kubectl expose deployment my-deploy --port=80 --target-port=8080

                                        6.2 网络排查#

kubectl get endpoints my-service

# 检查网络策略
kubectl get networkpolicies

# 测试 Service DNS 解析
kubectl run test-dns --image=busybox --rm -it -- nslookup my-service

# 检查节点网络
kubectl get nodes -o wide

                                        6.3 Ingress 管理#

kubectl get ingress

kubectl describe ingress my-ingress

kubectl get pods -n ingress-nginx

                                        7. Namespace / Node / 集群级命令#

                                        7.1 Namespace 管理#

                                        # 查看所有命名空间
kubectl get namespaces
kubectl get ns

# 创建命名空间
kubectl create namespace my-namespace

# 删除命名空间
kubectl delete namespace my-namespace

# 切换到指定命名空间
kubectl config set-context --current --namespace=my-namespace

# 查看命名空间资源配额
kubectl describe namespace my-namespace

                                        7.2 Node 管理#

                                        # 查看节点信息
kubectl get nodes -o wide

# 查看节点详细信息
kubectl describe node my-node

# 查看节点资源使用
kubectl top nodes

# 给节点打标签
kubectl label nodes my-node disktype=ssd

# 标记节点不可调度
kubectl cordon my-node

# 排空节点（迁移 Pod）
kubectl drain my-node --ignore-daemonsets --delete-emptydir-data

# 恢复节点调度
kubectl uncordon my-node

                                        7.3 集群级信息#

                                        # 查看集群组件状态
kubectl get pods -n kube-system

# 查看集群事件
kubectl get events -A --sort-by=.lastTimestamp

# 查看资源使用情况
kubectl top pods -A
kubectl top nodes

# 查看存储类
kubectl get storageclass

                                        8. 调试与排障常用命令#

                                        8.1 基础排障命令#

                                        # 查看 Pod 事件
kubectl describe pod my-pod | grep -A 10 Events

# 查看集群事件（按时间排序）
kubectl get events -A --sort-by=.metadata.creationTimestamp

# 查看 Pod 状态详情
kubectl get pod my-pod -o yaml

# 检查资源配额
kubectl describe resourcequota

                                        8.2 高级调试#

                                        # 使用临时调试容器
kubectl debug my-pod -it --image=busybox --target=my-pod

# 复制文件到 Pod
kubectl cp /local/path/file.txt my-pod:/path/in/pod/

# 从 Pod 复制文件
kubectl cp my-pod:/path/in/pod/file.txt /local/path/file.txt

# 检查 API 资源状态
kubectl get --raw='/readyz?verbose'

                                        8.3 性能监控#

                                        # 查看 Pod 资源使用
kubectl top pods

# 查看节点资源使用
kubectl top nodes

# 查看指定命名空间的资源使用
kubectl top pods -n my-namespace

# 持续监控资源使用
watch kubectl top pods

                                        8.4 选择器查询#

                                        # 使用标签选择器
kubectl get pods -l app=my-app

# 使用字段选择器
kubectl get pods --field-selector status.phase=Running

# 复杂标签查询
kubectl get pods -l 'app in (app1,app2)'

# 查看不健康的 Pod
kubectl get pods --field-selector status.phase!=Running

                                        9. 高级用法与技巧#

                                        9.1 标签和注解管理#

                                        # 添加标签
kubectl label pods my-pod environment=production

# 更新标签
kubectl label pods my-pod environment=staging --overwrite

# 删除标签
kubectl label pods my-pod environment-

# 添加注解
kubectl annotate pods my-pod description="My application pod"

# 查看标签和注解
kubectl get pods --show-labels

                                        9.2 高级输出格式#

                                        # JSONPath 查询
kubectl get pods -o jsonpath='{.items[*].metadata.name}'

# 自定义列输出
kubectl get pods -o custom-columns="NAME:.metadata.name,STATUS:.status.phase,NODE:.spec.nodeName"

# 使用 JSONPath 模板
kubectl get pods -o jsonpath='{range .items[*]}{.metadata.name}{"\t"}{.status.podIP}{"\n"}{end}'

# 提取特定信息
kubectl get pods -o jsonpath='{.items[0].metadata.name}'

                                        9.3 模板和导出#

                                        # 导出资源为 YAML（不包含状态信息）
kubectl get deployment my-deploy -o yaml --export > deployment.yaml

# 生成部署模板
kubectl create deployment my-deploy --image=nginx --dry-run=client -o yaml

# 生成服务模板
kubectl expose deployment my-deploy --port=80 --dry-run=client -o yaml

# 导出当前命名空间所有资源
kubectl get all -o yaml > all-resources.yaml

                                        9.4 组合命令和脚本#

                                        # 批量重启 Deployment
kubectl get deployments -o name | xargs -I {} kubectl rollout restart {}

# 查找所有运行失败的 Pod
kubectl get pods -A --field-selector=status.phase=Failed

# 查看 Pod 重启次数排名
kubectl get pods -A --sort-by='.status.containerStatuses[0].restartCount'

# 清理已完成的 Job
kubectl get jobs -o jsonpath='{.items[?(@.status.succeeded==1)].metadata.name}' | xargs kubectl delete job

# 监控 Pod 创建事件
kubectl get events -w --field-selector reason=Created

                                        10. 小场景示例：Pod 启动失败排查流程#

                                        排查 Pod 启动失败的完整流程#

                                        # 1. 查看 Pod 状态
kubectl get pods -n my-namespace

# 2. 查看 Pod 详细信息（重点关注 Events 部分）
kubectl describe pod my-pod -n my-namespace

# 3. 查看 Pod 日志
kubectl logs my-pod -n my-namespace

# 4. 如果容器之前崩溃过，查看之前容器的日志
kubectl logs my-pod -n my-namespace --previous

# 5. 检查相关事件
kubectl get events -n my-namespace --sort-by=.lastTimestamp

# 6. 检查节点资源
kubectl describe node $(kubectl get pod my-pod -o jsonpath='{.spec.nodeName}')

# 7. 检查资源配额
kubectl describe resourcequota -n my-namespace

# 8. 进入 Pod 调试（如果 Pod 正在运行但行为异常）
kubectl exec -it my-pod -n my-namespace -- /bin/sh

# 9. 检查服务依赖
kubectl get services -n my-namespace
kubectl describe service my-service -n my-namespace

# 10. 检查配置映射和密钥
kubectl get configmaps,secrets -n my-namespace
kubectl describe configmap my-config -n my-namespace

                                        部署更新问题排查#

                                        # 1. 查看部署状态
kubectl get deployments -n my-namespace

# 2. 查看滚动更新状态
kubectl rollout status deployment/my-deploy -n my-namespace

# 3. 查看更新历史
kubectl rollout history deployment/my-deploy -n my-namespace

# 4. 查看 ReplicaSet
kubectl get replicasets -n my-namespace -l app=my-app

# 5. 如果更新失败，回滚到上一个版本
kubectl rollout undo deployment/my-deploy -n my-namespace

# 6. 查看回滚状态
kubectl rollout status deployment/my-deploy -n my-namespace
