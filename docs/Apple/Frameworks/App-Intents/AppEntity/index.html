              This article is extracted from the chat log with AI. Please identify it with caution.

              AppEntity 是 App Intents 框架中的一个协议，用来让系统（如 Siri、快捷指令、Spotlight 等）识别你 App 中的某一类对象，比如：

              • 任务（Task）

              • 待办事项（ToDo）

              • 计时器（Timer）

              • 音乐、相册、联系人……

              • 你 App 中任何可操作的“实体”

              换句话说，它是让你的 App 中的「对象」能成为 Siri / 快捷指令的输入和输出单位的桥梁。

              🧱 AppEntity 的结构 #

              struct Task: AppEntity {
    var id: String
    var title: String

    static var typeDisplayRepresentation: TypeDisplayRepresentation {
        "Task"
    }

    var displayRepresentation: DisplayRepresentation {
        DisplayRepresentation(title: title)
    }

    static var defaultQuery = TaskQuery()
}

              必须实现的核心部分：

              部分说明
              id实体的唯一标识
              typeDisplayRepresentationSiri 显示这个类型用什么名字（比如“任务”、“计时器”）
              displayRepresentation显示具体某个实例时的名称（比如“写日报”）
              defaultQuery提供搜索、选择功能的数据源入口（一般是一个查询器）

              🧪 示例：你的计时器项目中加入 AppEntity #

              Step 1：定义 Timer 实体 #

              import AppIntents

struct FocusTimer: AppEntity, Identifiable {
    var id: String
    var name: String
    var duration: Int

    static var typeDisplayRepresentation: TypeDisplayRepresentation {
        "Focus Timer"
    }

    var displayRepresentation: DisplayRepresentation {
        DisplayRepresentation(title: "\(name) (\(duration) min)")
    }

    static var defaultQuery = FocusTimerQuery()
}

              Step 2：定义实体查询器 #

              你必须提供一个 EntityQuery 来告诉系统有哪些可选对象可以让用户挑选。

              struct FocusTimerQuery: EntityQuery {
    func entities(for identifiers: [FocusTimer.ID]) async throws -> [FocusTimer] {
        // 通过 id 查找实体（通常来自你的数据库）
        sampleTimers.filter { identifiers.contains($0.id) }
    }

    func suggestedEntities() async throws -> [FocusTimer] {
        // 默认展示的候选项
        sampleTimers
    }

    var sampleTimers: [FocusTimer] {
        [
            FocusTimer(id: "1", name: "写作", duration: 25),
            FocusTimer(id: "2", name: "读书", duration: 15),
            FocusTimer(id: "3", name: "思考", duration: 30)
        ]
    }
}

              Step 3：让 Intent 使用这个实体 #

              struct StartNamedTimerIntent: AppIntent {
    static var title: LocalizedStringResource = "Start Named Timer"

    @Parameter(title: "Focus Session")
    var timer: FocusTimer

    func perform() async throws -> some IntentResult {
        TimerController.shared.startTimer(for: timer.duration)
        return .result(dialog: "Started \(timer.name) for \(timer.duration) minutes.")
    }
}

              系统会自动提供一个“选择计时器”的 UI，让用户选择 写作 / 读书 等任务。

              🧩 步骤三：创建 AppIntent（StartNamedTimerIntent.swift） #

              import AppIntents
import ActivityKit

struct StartNamedTimerIntent: AppIntent {
    static var title: LocalizedStringResource = "Start Named Focus Timer"
    static var description = IntentDescription("Start a specific focus timer by name and show it in the Dynamic Island.")

    @Parameter(title: "Which Focus Timer")
    var timer: FocusTimer

    func perform() async throws -> some IntentResult {
        let duration = TimeInterval(timer.duration * 60)
        let attributes = TimerAttributes(taskName: timer.name)
        let contentState = TimerAttributes.ContentState(remainingTime: duration)

        do {
            _ = try Activity<TimerAttributes>.request(attributes: attributes, contentState: contentState)
            return .result(dialog: "已启动「\(timer.name)」计时器，共 \(timer.duration) 分钟")
        } catch {
            return .result(dialog: "启动失败，请稍后重试")
        }
    }
}

              🧩 一、Live Activity 倒计时自动更新 #

              我们需要持续更新 Activity 的 remainingTime，你可以使用定时器或后台任务管理。

              修改 TimerAttributes.ContentState #

              你已有：

              struct ContentState: Codable, Hashable {
    var remainingTime: TimeInterval
}

              添加一个后台控制器类：TimerController.swift #

              import ActivityKit

class TimerController: ObservableObject {
    static let shared = TimerController()

    private var activity: Activity<TimerAttributes>?
    private var timer: Timer?

    func startActivity(taskName: String, duration: TimeInterval) {
        let attr = TimerAttributes(taskName: taskName)
        let state = TimerAttributes.ContentState(remainingTime: duration)

        do {
            activity = try Activity<TimerAttributes>.request(attributes: attr, contentState: state)
            startCountdown(duration: duration)
        } catch {
            print("启动 Live Activity 失败：\(error)")
        }
    }

    func startCountdown(duration: TimeInterval) {
        var remaining = duration
        timer?.invalidate()

        timer = Timer.scheduledTimer(withTimeInterval: 1, repeats: true) { _ in
            remaining -= 1
            Task {
                await self.activity?.update(using: .init(remainingTime: remaining))
            }
            if remaining <= 0 {
                self.endActivity()
            }
        }
    }

    func endActivity() {
        timer?.invalidate()
        Task {
            await activity?.end(using: .init(remainingTime: 0), dismissalPolicy: .immediate)
        }
    }
}

struct StopTimerIntent: AppIntent {

    private let key = "completedTimers"

    init() {
        load()
    }

        let record = CompletedTimer(name: name, duration: duration, completedAt: .now)
        history.insert(record, at: 0)
        save()

    private func load() {
        if let data = UserDefaults.standard.data(forKey: key),
           let decoded = try? JSONDecoder().decode([CompletedTimer].self, from: data) {
            self.history = decoded
        }
    }

    private func save() {
        if let data = try? JSONEncoder().encode(history) {
            UserDefaults.standard.set(data, forKey: key)
        }
    }
}

              🧩 二、记录 iCloud 同步功能（基础实现） #

              修改 TimerHistoryStore 实现 #

              我们将从 UserDefaults 切换为 NSUbiquitousKeyValueStore，以启用 iCloud 同步。

              替换保存与读取逻辑 #

              import Foundation
import Combine

class TimerHistoryStore: ObservableObject {
    static let shared = TimerHistoryStore()
    private let key = "completedTimers"
    private let store = NSUbiquitousKeyValueStore.default

    @Published private(set) var history: [CompletedTimer] = []

    init() {
        load()
        NotificationCenter.default.addObserver(
            self,
            selector: #selector(iCloudDidChange(_:)),
            name: NSUbiquitousKeyValueStore.didChangeExternallyNotification,
            object: store
        )
        store.synchronize()
    }

    func addRecord(name: String, duration: Int) {
        let record = CompletedTimer(name: name, duration: duration, completedAt: .now)
        history.insert(record, at: 0)
        save()
    }

    @objc private func iCloudDidChange(_ notification: Notification) {
        load()
    }

    private func load() {
        if let data = store.data(forKey: key),
           let decoded = try? JSONDecoder().decode([CompletedTimer].self, from: data) {
            DispatchQueue.main.async {
                self.history = decoded
            }
        }
    }

    private func save() {
        if let data = try? JSONEncoder().encode(history) {
            store.set(data, forKey: key)
            store.synchronize()
        }
    }
}

              extension TimerHistoryStore {
    func exportCSV() -> String {
        var csv = "名称,时长(分钟),完成时间\n"
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM-dd HH:mm"

        for record in history {
            let line = "\"\(record.name)\",\(record.duration),\"\(formatter.string(from: record.completedAt))\""
            csv.append(line + "\n")
        }
        return csv
    }
}

              🧩 三、支持导出为文档类型 #

              创建 CSVDocument.swift（仅用于 fileExporter） #

              import SwiftUI
import UniformTypeIdentifiers

struct CSVDocument: FileDocument {
    static var readableContentTypes: [UTType] { [.commaSeparatedText] }

    var data: Data

    init(data: Data) {
        self.data = data
    }

    init(configuration: ReadConfiguration) throws {
        guard let data = configuration.file.regularFileContents else {
            throw CocoaError(.fileReadCorruptFile)
        }
        self.data = data
    }

    func fileWrapper(configuration: WriteConfiguration) throws -> FileWrapper {
        return .init(regularFileWithContents: data)
    }
}
